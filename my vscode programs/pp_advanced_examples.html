C++ Advanced Examples

Examples (1-70)

1. Class and Constructor


    #include <iostream>
    using namespace std;

    class User {
    public:
        string name;
        User(string name) : name(name) {}
    };

    int main() {
        User user("Alice");
        cout << user.name << endl;
        return 0;
    }

2. Smart Pointer


    #include <memory>
    #include <iostream>

    using namespace std;

    int main() {
        unique_ptr ptr = make_unique(10);
        cout << *ptr << endl;
        return 0;
    }

3. Lambda Functions


    #include <iostream>
    using namespace std;

    int main() {
        auto square = [](int x) { return x * x; };
        cout << square(5) << endl;
        return 0;
    }

70. Final Class


    #include <iostream>
    using namespace std;

    class FinalClass final {
    public:
        void display() { cout << "Final Class Example"; }
    };

    int main() {
        FinalClass fc;
        fc.display();
        return 0;
    }
