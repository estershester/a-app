CSS Advanced Examples

Examples (1-100)

1. Gradient Background


.gradient {
    background: linear-gradient(to right, #ff7e5f, #feb47b);
}

2. Box Shadow


.box-shadow {
    box-shadow: 0 4px 8px rgba(0, 0, 0, 0.3);
}

3. Flexbox Layout


.flex-container {
    display: flex;
    justify-content: space-around;
    align-items: center;
}

4. Grid Layout


.grid-container {
    display: grid;
    grid-template-columns: repeat(3, 1fr);
    gap: 10px;
}

5. CSS Animation


@keyframes bounce {
    0%, 100% { transform: translateY(0); }
    50% { transform: translateY(-50px); }
}

.bounce {
    animation: bounce 2s infinite;
}

6. Hover Effect


.hover-effect {
    transition: transform 0.3s;
}
.hover-effect:hover {
    transform: scale(1.1);
}

7. Sticky Header


.sticky-header {
    position: sticky;
    top: 0;
    background: #333;
    color: #fff;
}

8. Responsive Design


@media (max-width: 600px) {
    .responsive {
        width: 100%;
        background: lightblue;
    }
}

9. Glassmorphism Effect


.glass {
    backdrop-filter: blur(10px);
    background: rgba(255, 255, 255, 0.2);
    border-radius: 10px;
}

10. Tooltip


.tooltip {
    position: relative;
    display: inline-block;
}
.tooltip::after {
    content: 'Tooltip text';
    position: absolute;
    visibility: hidden;
    background: #000;
    color: #fff;
    padding: 5px;
    border-radius: 5px;
}
.tooltip:hover::after {
    visibility: visible;
}

11. CSS Variables


:root {
    --main-color: #6c5ce7;
}
.variable {
    color: var(--main-color);
}

12. Parallax Effect


.parallax {
    background: url('image.jpg') no-repeat center center fixed;
    background-size: cover;
    height: 400px;
}

13. Scrollbar Styling


::-webkit-scrollbar {
    width: 10px;
}
::-webkit-scrollbar-thumb {
    background: #888;
}
::-webkit-scrollbar-thumb:hover {
    background: #555;
}

14. CSS Counters


.counter {
    counter-reset: section;
}
.counter p::before {
    counter-increment: section;
    content: "Section " counter(section) ": ";
}

15. Grid Auto-Fit


.auto-grid {
    display: grid;
    grid-template-columns: repeat(auto-fit, minmax(100px, 1fr));
}

16. Text Shadow


.text-shadow {
    text-shadow: 2px 2px 5px #000;
}

17. Pseudo-Elements


.pseudo::before {
    content: "Before Text - ";
}
.pseudo::after {
    content: " - After Text";
}

18. Clip Path


.clip-path {
    clip-path: circle(50% at center);
}

19. CSS Masking


.mask {

}

20. 3D Transform


.three-d {

}

21. Fixed Footer


.fixed-footer {

100. Animated Loader


.loader {
    width: 50px;
    height: 50px;
    border: 5px solid #f3f3f3;
    border-top: 5px solid #3498db;
    border-radius: 50%;
    animation: spin 1s linear infinite;
}
@keyframes spin {
    to {
        transform: rotate(360deg);
    }
}
