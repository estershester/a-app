2. Async/Await


async function fetchData() {
    let res = await fetch('/api');
    let data = await res.json();
    console.log(data);
}

11. Inheritance

        console.log('Animal sound');
    }
}

20. Debouncing


function debounce(func, delay) {
    let timer;
    return (...args) => {
        clearTimeout(timer);
        timer = setTimeout(() => func(...args), delay);
    };
}
window.addEventListener('resize', debounce(() => {
    console.log('Resized!');
}, 300));

23. Proxy Object
