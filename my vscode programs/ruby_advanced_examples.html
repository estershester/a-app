Ruby Advanced Examples

Examples (1-70)

1. Class and Object


class User
  attr_accessor :name

  def initialize(name)
    @name = name
  end
end

user = User.new("Alice")
puts user.name

2. Singleton Pattern


require 'singleton'

class Logger
  include Singleton

  def log(message)
    puts message
  end
end

Logger.instance.log("Singleton Example")

3. Blocks and Procs


square = Proc.new { |x| x * x }
puts square.call(5)

4. Lambda


square = ->(x) { x * x }
puts square.call(5)

5. Exception Handling


begin
  raise "An error occurred"
rescue => e
  puts e.message
end

6. File Handling


File.open('example.txt', 'w') { |file| file.write("Hello, File!") }

7. JSON Parsing


require 'json'
data = '{"name": "Alice", "age": 25}'
parsed = JSON.parse(data)
puts parsed["name"]

8. REST API Call


require 'net/http'
require 'uri'

uri = URI.parse('https://jsonplaceholder.typicode.com/posts/1')
response = Net::HTTP.get(uri)
puts response

9. Module


module Greeting
  def say_hello
    puts "Hello!"
  end
end

class User
  include Greeting
end

user = User.new
user.say_hello

10. Symbol Example


person = { name: "Alice", age: 25 }
puts person[:name]

11. Hash Iteration


hash = { a: 1, b: 2, c: 3 }
hash.each { |key, value| puts "#{key}: #{value}" }

12. Array Map


numbers = [1, 2, 3]
squared = numbers.map { |n| n * n }
puts squared

13. Reduce Method


sum = [1, 2, 3].reduce(:+)
puts sum

14. Threads


Thread.new { puts "Thread Example" }.join

15. Enumerator


enum = [1, 2, 3].each
puts enum.next

16. Class Inheritance


class Animal
  def speak
    puts "Animal speaks"
  end
end

class Dog < Animal
  def speak
    puts "Dog barks"
  end
end

Dog.new.speak

17. Duck Typing


def make_sound(animal)
  animal.speak
end

class Cat
  def speak
    puts "Meow"
  end
end

make_sound(Cat.new)

18. Regular Expressions


text = "The price is $100"
puts text.match(/\$\d+/)[0]

19. Date and Time


puts Time.now.strftime("%Y-%m-%d %H:%M:%S")
